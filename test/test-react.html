Gallop Player - React Integration

The React wrapper uses <GallopPlayer /> which directly instantiates GallopPlayerCore on a div ref (no Web Component interop). 

import { GallopPlayer } from '@scalemule/gallop/react';

function VideoPage() {
  return (
    <GallopPlayer
      src="https://test-streams.mux.dev/x36xhzz/x36xhzz.m3u8"
      controls
      onPlay={() => console.log('Playing')}
      onPause={() => console.log('Paused')}
      onError={(e) => console.error(e.code, e.message)}
      onReady={() => console.log('Player ready')}
      style={{ maxWidth: 800 }}
    />
  );
}
// With ref for imperative control
import { useRef } from 'react';
import { GallopPlayer, GallopPlayerHandle } from '@scalemule/gallop/react';

function VideoWithControls() {
  const ref = useRef<GallopPlayerHandle>(null);

  return (
    <div>
      <GallopPlayer ref={ref} src="..." />
      <button onClick={() => ref.current?.play()}>Play</button>
      <button onClick={() => ref.current?.seek(30)}>Skip to 30s</button>
    </div>
  );
}
// Hook-based usage
import { useGallopPlayer } from '@scalemule/gallop/react';

function CustomPlayer() {
  const { containerRef, getPlayer } = useGallopPlayer({
    src: 'https://example.com/video.m3u8',
    apiKey: 'sm_live_xxx',
    controls: true,
  });

  return <div ref={containerRef} style={{ maxWidth: 800 }} />;
}
